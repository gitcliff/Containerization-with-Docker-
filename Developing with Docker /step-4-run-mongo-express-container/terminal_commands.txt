# üì¶ SUMMARY  
# These commands run a Mongo Express container connected to the MongoDB container on a custom Docker network, expose its web UI port, and check container status and logs.  
# Language/Platform: Bash (Docker CLI)  
# What you'll learn: How to launch Mongo Express configured to connect to MongoDB securely with admin credentials and view the web interface logs  
# Outcome: A running Mongo Express container accessible at http://localhost:8081, connected to your MongoDB container for database management via a web UI  
# Why this matters: Provides a convenient, user-friendly web interface to manage MongoDB data without using command line tools  

# ------------------------------  
# üß© SECTION TITLE: Run Mongo Express Container Connected to MongoDB  
# ------------------------------  

docker run --network mongo-network --name mongo-express \
  -e ME_CONFIG_MONGODB_SERVER=mongodb \
  -e ME_CONFIG_MONGODB_ADMINUSERNAME=admin \
  -e ME_CONFIG_MONGODB_ADMINPASSWORD=pass \
  -e ME_CONFIG_MONGODB_URL=mongodb://admin:pass@mongodb:27017/ \
  -p 8081:8081 \
  -d mongo-express
# ‚úÖ What it does:  
# - Runs the mongo-express container in detached mode (`-d`) so it runs in the background  
# - Connects the container to the custom Docker network `mongo-network` where your MongoDB container lives  
# - Passes MongoDB connection settings via environment variables (`-e` flags):  
#   - Server hostname (`ME_CONFIG_MONGODB_SERVER=mongodb`)  
#   - Admin username & password (`ME_CONFIG_MONGODB_ADMINUSERNAME`, `ME_CONFIG_MONGODB_ADMINPASSWORD`)  
#   - Full MongoDB URI (`ME_CONFIG_MONGODB_URL`) for fallback connection  
# - Maps host port 8081 to container port 8081 (`-p 8081:8081`) so you can access the web UI at http://localhost:8081  
# - Names this container `mongo-express` for easy reference  
# üß† Why it matters: Sets up Mongo Express to talk securely to your MongoDB instance and exposes it for browser-based management  

docker ps  
# ‚úÖ What it does: Lists running containers to verify `mongo-express` is active  
# üß† Why it matters: Confirms the container started successfully and is ready for web access  

docker logs mongo-express  
# ‚úÖ What it does: Shows startup logs for the `mongo-express` container  
# üß† Why it matters: Useful for troubleshooting connection errors or misconfigured environment variables
