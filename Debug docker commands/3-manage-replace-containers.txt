# ------------------------------  
# üß© SECTION TITLE: Stop and Start Redis Containers with Custom Names  
# ------------------------------  
docker stop amazing_chatterjee  
# ‚úÖ What it does: Stops the running container named `amazing_chatterjee` gracefully  
# üß† Why it matters: Frees system resources and avoids port conflicts before starting a new container  

docker run -d -p 6000:6379 --name redis-latest redis  
# ‚úÖ What it does:  
# - Runs the latest Redis container in detached mode  
# - Maps host port 6000 to container port 6379  
# - Names the container `redis-latest` for easier management  
# üß† Why it matters: You can access Redis on localhost:6000 and manage the container by name instead of ID  

docker ps  
# ‚úÖ What it does: Lists all running containers including `redis-latest`  
# üß† Why it matters: Confirms your new Redis container is up and running with the expected configuration
